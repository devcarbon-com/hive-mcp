\begin{frame}[label={sec:orgca5bd7a},fragile]{Usage Example}
 \begin{block}{Parse and Query}
\begin{verbatim}
(require '[hive-mcp.org-clj.parser :as p])
(require '[hive-mcp.org-clj.query :as q])

(def doc (p/parse-document (slurp "tasks.org")))

;; Find all TODOs
(q/find-by-status doc "TODO")

;; Get statistics
(q/task-stats doc)
;; => {:total 10 :todo 3 :done 7}
\end{verbatim}
\end{block}
\end{frame}

\section{Getting Started}
\label{sec:orgc7970fa}
\subsection{Installation}
\label{sec:orgd3b8198}
\begin{frame}[label={sec:org2bd95ab},fragile]{Setup Steps}
 \begin{block}{1. Clone Repository}
\begin{verbatim}
git clone https://github.com/BuddhiLW/hive-mcp.git
\end{verbatim}
\end{block}
\begin{block}{2. Configure Emacs}
\begin{verbatim}
;; Add to load-path
(add-to-list 'load-path "path/to/hive-mcp/elisp")
(add-to-list 'load-path "path/to/hive-mcp/elisp/addons")

;; Load main package
(require 'hive-mcp)
(require 'hive-mcp-addons)

;; Enable addons
(setq hive-mcp-addon-always-load '(cider org-kanban))
\end{verbatim}
\end{block}
\end{frame}
